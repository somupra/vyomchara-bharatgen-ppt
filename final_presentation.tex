\section{The Problem: An Industry Ripe for Disruption}

\begin{frame}{The Indian Mining Sector: A Foundation of the Economy Facing Foundational Cracks}
    \begin{itemize}[<+-| alert@+>] 
        \item \textbf{Technological Stagnation:} High resistance to new tech due to entrenched practices and workforce concerns.
        \item \textbf{Fragmented Solutions:} Lack of a single, end-to-end platform for the entire mining lifecycle (Auction -> Exploration -> Production -> Delivery).
        \item \textbf{Operational Inefficiency:} Challenges in key areas like ventilation, geological surveying, and resource management.
        \item \textbf{High Costs & Remote Infra:} Expensive equipment and limited infrastructure in key mining regions.
    \end{itemize}

\begin{frame}{Voxel: Lab-Grade Hyperspectral Analysis, Anywhere.}
    \begin{itemize}
        \item \textbf{Description:} A portable, drone-mountable sensor for real-time, non-destructive mineral analysis.
        \item \textbf{Key Features:}
            \begin{itemize}
                \item \textbf{Functionality:} Iron ore grade detection, mineral mapping.
                \item \textbf{Technology:} CNC body, advanced outsourced camera, ESP32 for control.
