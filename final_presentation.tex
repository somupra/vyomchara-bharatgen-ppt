\section{The Core Problem in Indian Mining}

\begin{frame}{A Foundational Industry Facing a Digital Divide}
    \begin{itemize}[<+-| alert@+>] 
        \item \textbf{The Challenge:} The primary issue is a **lack of integrated, end-to-end technological solutions.**
        \item \textbf{Why is it hard?}
            \begin{itemize}
                \item **Multiple Stakeholders:** Conflicting interests between workers, managers, and contractors.
                \item **Fragmented Technology:** Existing solutions are piecemeal and don't cover the full operational chain.
                \item **Remote & Cost-Sensitive:** Operations are in remote locations with limited infrastructure.
            \end{itemize}
    \end{itemize}

\section{Our Solution: The Vyomchara Platform}

\begin{frame}{Acquire, Analyze, Act: A Unified Platform for the Mine of the Future}
    \begin{itemize}
        \item \textbf{Core Concept:} We integrate hardware, software, and AI to provide a single source of truth for mining operations.
        \item \textbf{Three Pillars:}
            \begin{enumerate}
                \item \textbf{Acquire:} Use cost-effective drones, IoT, and our proprietary "Voxel" sensor.
                \item \textbf{Analyze:} Apply AI/ML models (SAM+CNN fusion) for precise mineral detection.
                \item \textbf{Act:} Provide a "Mining 4.0 Dashboard" for automation and management.
            \end{enumerate}
    \end{itemize}
